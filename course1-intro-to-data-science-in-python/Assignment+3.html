import pandas as pd 
import numpy as np
import re

def datasets():
    data = pd.read_excel('Energy Indicators.xls', skiprows=17, skip_footer=38)
    data.drop(data.columns[[0,1]], axis = 1, inplace = True)
    data.columns = ['Country', 'Energy Supply', 'Energy Supply per Capita', '% Renewable']
    data['Energy Supply'] = data['Energy Supply'].replace('...', np.NaN)
    data['Energy Supply'] *= 1000000
    data['Energy Supply per Capita'] = data['Energy Supply per Capita'].replace('...', np.NaN)
    
    data['Country'] = data['Country'].apply(lambda x:re.sub("\d+", "", x)) ## remove numbers
    data['Country'] = data['Country'].apply(lambda x:re.sub(r' \([^)]*\)', '', x)) ## remove text in ()
    
    replacing = {"Republic of Korea": "South Korea",
                 "United States of America": "United States",
                 "United Kingdom of Great Britain and Northern Ireland": "United Kingdom",
                 "China, Hong Kong Special Administrative Region": "Hong Kong",
                 "China, Macao Special Administrative Region": "Macao"
                }
    data['Country'].replace(replacing, inplace=True)
    
    # PART 2 #
    GDP = pd.read_csv('world_bank.csv', skiprows = 4)
    renaming = ({"Korea, Rep.": "South Korea", 
                  "Iran, Islamic Rep.": "Iran", 
                  "Hong Kong SAR, China": "Hong Kong"})
    GDP['Country Name'].replace(renaming, inplace=True)
    
    
    # PART 3 #
    ScimEn = pd.read_excel('scimagojr-3.xlsx')
    return data,GDP,ScimEn

def answer_one():
    data,GDP,ScimEn = datasets()
    merged = pd.merge(data, GDP, how='inner', left_on='Country', right_on='Country Name').merge(ScimEn, how='inner', left_on='Country', right_on='Country')
    merged.set_index('Country', inplace=True)
    merged = merged[['Rank', 'Documents', 'Citable documents', 'Citations', 
                   'Self-citations', 'Citations per document', 'H index', 'Energy Supply', 
                   'Energy Supply per Capita', '% Renewable', '2006', '2007', '2008', '2009', 
                   '2010', '2011', '2012', '2013', '2014', '2015']].sort_values('Rank', ascending=True)
    return merged.head(15)
